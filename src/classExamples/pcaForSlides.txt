myT <- read.table("nc101_scaff_dataCounts.txt", header=TRUE, sep="\t",row.names=1)

myTNorm <- myT

	colSum = sum(myT[,i])
	myTNorm[,i] = log10( myTNorm[,i]/colSum + 0.000001)
}
