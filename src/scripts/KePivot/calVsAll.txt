rm(list=ls())

setwd("C:\\kePivot")

myT <- read.table("genusAsColumnsLogNormPlusMeta.txt", sep="\t", header=TRUE)

pValues <- vector()
taxaName <- vector()

index <- 1
for ( i in 3:ncol(myT)) 
{
	if( sum(myT[,i])/nrow(myT) > 1 ) 
	{
		aLm <- lm( myT[,2] ~ myT[,i])
		pValues[index] = anova(aLm)$"Pr(>F)"[1]
		taxaName[index] = names(myT)[i]
		index = index + 1
	}
	
}
