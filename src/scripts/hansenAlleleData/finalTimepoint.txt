myT <- read.table("E_coli_NC101ForR.txt",header=TRUE,sep="\t")

pValues <- vector()

index <- 1
for( i in 1:nrow(myT))
{
		koVal <- myT[i, 8:16]
		wtVal <- myT[i,17:27]
		
		if( sum(! is.na( koVal  )) > 5 & sum(! is.na( wtVal )) > 5 ) 
		{
			pValues[index] <- t.test( koVal,  wtVal)$p.value
			index= index + 1
		}
}
